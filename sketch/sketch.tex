\section{Fert-Simplex} 
\begin{align}
	& \text{Min} \quad
	 \frac{\sum_{i=1}^{n}a_{i1}x_1}{p_1} +
	 \frac{\sum_{i=1}^{n}a_{i2}x_2}{p_2} +
	 \frac{\sum_{i=1}^{n}a_{i3}x_3}{p_3} \\
	  \quad \text{s.t.} \\
	& \quad 2x_1 + x_3 \geq 1 \\
	& \quad x_2 + x_3 \geq 5 \\
	& \quad x_1 + 3x_2 \geq 3 \\
	& \quad x_2 \geq 11 \\
	& \quad x_3 \geq 8 \\
\end{align}
\label{eq:Fert-Simplex minimization}
